```
3. Kiterunner  (if opts.stages.kiterunner — default false, opt-in per DISC-05)
4. httpx       (if opts.stages.httpx — default true; probes ALL endpoints accumulated so far)
   - Pass 1: unauth → fill requiresAuth tri-valor

5. Arjun       (if opts.stages.arjun — default false AND arjunEndpointIds required)

```

**Create `server/services/journeys/apiDiscovery.ts`**. This is the largest file in the phase (~300 lines). Structure it as:

```typescript
// Phase 11 — API Discovery & Enrichment orchestrator.
// Sequences spec → crawler → kiterunner → httpx → arjun per CONTEXT.md.
// Each stage fails independently (log + skip); pipeline continues.
// Resultados parciais persistem em cancel.
import type { Api, ApiEndpoint, InsertApiEndpoint } from '@shared/schema';
import type { DiscoverApiOpts } from '@shared/schema';
import { createLogger } from '../../lib/logger';
import { storage } from '../../storage';
import { fetchAndParseSpec, specToEndpoints } from '../scanners/api/openapi';
import { probeGraphQL, schemaToEndpoints } from '../scanners/api/graphql';
import { runKatana, type KatanaCredential } from '../scanners/api/katana';
import { runKiterunner } from '../scanners/api/kiterunner';
import { runHttpx, mapRequiresAuth } from '../scanners/api/httpx';
import { runArjun } from '../scanners/api/arjun';

const log = createLogger('journeys:apiDiscovery');

const ARJUN_WORDLIST = '/opt/samureye/wordlists/arjun-extended-pt-en.txt';

export interface DiscoveryResult {
  apiId: string;
  stagesRun: Array<'spec' | 'crawler' | 'kiterunner' | 'httpx' | 'arjun'>;
  stagesSkipped: Array<{ stage: string; reason: string }>;
  endpointsDiscovered: number;
  endpointsUpdated: number;
  endpointsStale: string[];
  specFetched?: {
    url: string;
    version: string;
    hash: string;
    driftDetected: boolean;
  };
  cancelled: boolean;
  durationMs: number;
}

export async function discoverApi(
  apiId: string,
  opts: DiscoverApiOpts,
  jobId?: string,
): Promise<DiscoveryResult> {
  const startedAt = Date.now();
  const signal = getAbortSignalForJob(jobId);
  const stagesRun: DiscoveryResult['stagesRun'] = [];
  const stagesSkipped: DiscoveryResult['stagesSkipped'] = [];
  let endpointsDiscovered = 0;
  let endpointsUpdated = 0;
  let specFetched: DiscoveryResult['specFetched'] | undefined;
  let cancelled = false;

  const api = await storage.getApi(apiId);
  if (!api) {
    throw new Error(`API não encontrada: ${apiId}`);
  }

  // dryRun forces off crawler/kiterunner/arjun — explicit override semantics.
  const effectiveStages = {
    spec: opts.stages.spec && !opts.dryRun ? opts.stages.spec : (opts.dryRun ? opts.stages.spec : opts.stages.spec),
    crawler: opts.dryRun ? false : opts.stages.crawler,
    kiterunner: opts.dryRun ? false : opts.stages.kiterunner,
    httpx: opts.stages.httpx,
    arjun: opts.dryRun ? false : opts.stages.arjun,
  };
  // (fix the spec branch — dryRun keeps spec on)
  effectiveStages.spec = opts.stages.spec;

  log.info({ apiId, jobId, baseUrl: api.baseUrl, apiType: api.apiType, stages: effectiveStages, dryRun: opts.dryRun }, 'discoverApi started');

  // --- STAGE: SPEC ---
  if (effectiveStages.spec) {
    try {
      const result = await runSpecStage(api, opts, signal);
      if (result) {
        stagesRun.push('spec');
        specFetched = result.specFetched;
        const counts = await storage.upsertApiEndpoints(apiId, result.endpoints);
        endpointsDiscovered += counts.inserted;
        endpointsUpdated += counts.updated;
      } else {
        stagesSkipped.push({ stage: 'spec', reason: 'no spec found' });
      }
    } catch (err) {
      log.error({ err: String(err), apiId, stage: 'spec' }, 'stage failed');
      stagesSkipped.push({ stage: 'spec', reason: String(err) });
    }
  } else {
    stagesSkipped.push({ stage: 'spec', reason: 'disabled' });
  }

  if (signal.aborted) { cancelled = true; return finalize(); }

  // --- STAGE: CRAWLER (Katana) ---
  if (effectiveStages.crawler) {
    try {
      const credential = await buildKatanaCredential(apiId, api.baseUrl, opts.credentialIdOverride);
      const result = await runKatana(api.baseUrl, { depth: opts.katana?.depth, headless: opts.katana?.headless, credential }, { apiId, jobId, signal });
      if (result.skipped) {
        stagesSkipped.push({ stage: 'crawler', reason: result.skipped.reason });
      } else {
        stagesRun.push('crawler');
        const counts = await storage.upsertApiEndpoints(apiId, result.endpoints);
        endpointsDiscovered += counts.inserted;
        endpointsUpdated += counts.updated;
      }
    } catch (err) {
      log.error({ err: String(err), apiId, stage: 'crawler' }, 'stage failed');
      stagesSkipped.push({ stage: 'crawler', reason: String(err) });
    }
  } else {
    stagesSkipped.push({ stage: 'crawler', reason: opts.dryRun ? 'dryRun' : 'disabled' });
  }

  if (signal.aborted) { cancelled = true; return finalize(); }

  // --- STAGE: KITERUNNER (opt-in) ---
  if (effectiveStages.kiterunner) {
    try {
      const result = await runKiterunner(api.baseUrl, { rateLimit: opts.kiterunner?.rateLimit }, { apiId, jobId, signal });
      if (result.skipped) {
        stagesSkipped.push({ stage: 'kiterunner', reason: result.skipped.reason });
      } else {
        stagesRun.push('kiterunner');
        const counts = await storage.upsertApiEndpoints(apiId, result.endpoints);
        endpointsDiscovered += counts.inserted;
        endpointsUpdated += counts.updated;
      }
    } catch (err) {
      log.error({ err: String(err), apiId, stage: 'kiterunner' }, 'stage failed');
      stagesSkipped.push({ stage: 'kiterunner', reason: String(err) });
    }
  } else {
    stagesSkipped.push({ stage: 'kiterunner', reason: opts.dryRun ? 'dryRun' : 'disabled (opt-in)' });
  }

  if (signal.aborted) { cancelled = true; return finalize(); }

  // --- STAGE: HTTPX (2 passes) ---
  let allEndpoints: ApiEndpoint[] = [];
  if (effectiveStages.httpx) {
    try {
      allEndpoints = await storage.listEndpointsByApi(apiId);
      const urls = allEndpoints.map((e) => buildFullUrl(api.baseUrl, e.path));
      const pass1 = await runHttpx(urls, {}, { jobId, signal });
      if (pass1.skipped) {
        stagesSkipped.push({ stage: 'httpx', reason: pass1.skipped.reason });
      } else {
        stagesRun.push('httpx');
        // Merge pass-1 results; also update requiresAuth via direct upsert path.
        await mergeHttpxResultsIntoEndpoints(allEndpoints, pass1.results);

        // Pass 2: auth probe for endpoints flagged requiresAuth=true with compatible cred.
        const authUrls: Array<{ url: string; endpointId: string; authHeader: string }> = [];
        for (const ep of allEndpoints) {
          const pr = pass1.results.find((r) => r.inputUrl === buildFullUrl(api.baseUrl, ep.path) || r.url === buildFullUrl(api.baseUrl, ep.path));
          if (pr?.requiresAuth === true) {
            const authHeader = await buildHttpxAuthHeader(apiId, buildFullUrl(api.baseUrl, ep.path), opts.credentialIdOverride);
            if (authHeader) authUrls.push({ url: buildFullUrl(api.baseUrl, ep.path), endpointId: ep.id, authHeader });
          }
        }
        if (authUrls.length > 0) {
          // httpx can only take one header per run; group by header (typically 1 for per-API cred).
          const grouped = new Map<string, string[]>();
          for (const item of authUrls) {
            if (!grouped.has(item.authHeader)) grouped.set(item.authHeader, []);
            grouped.get(item.authHeader)!.push(item.url);
          }
          for (const [authHeader, groupUrls] of grouped) {
            const pass2 = await runHttpx(groupUrls, { authHeader }, { jobId, signal });
            if (!pass2.skipped) {
              await mergeHttpxResultsIntoEndpoints(allEndpoints, pass2.results);
            }
          }
        }
      }
    } catch (err) {
      log.error({ err: String(err), apiId, stage: 'httpx' }, 'stage failed');
      stagesSkipped.push({ stage: 'httpx', reason: String(err) });
    }
  } else {
    stagesSkipped.push({ stage: 'httpx', reason: 'disabled' });
  }

  if (signal.aborted) { cancelled = true; return finalize(); }

  // --- STAGE: ARJUN (opt-in + user-selected) ---
  if (effectiveStages.arjun) {
    try {
      if (!opts.arjunEndpointIds || opts.arjunEndpointIds.length === 0) {
        throw new Error('arjunEndpointIds obrigatório quando stages.arjun=true');
      }
      if (allEndpoints.length === 0) allEndpoints = await storage.listEndpointsByApi(apiId);
      const endpointsById = new Map(allEndpoints.map((e) => [e.id, e]));
      const selected: ApiEndpoint[] = [];
      for (const id of opts.arjunEndpointIds) {
        const ep = endpointsById.get(id);
        if (!ep) throw new Error(`arjunEndpointId ${id} não existe em api_endpoints`);
        if (ep.apiId !== apiId) throw new Error(`arjunEndpointId ${id} pertence a apiId diferente`);
        if (ep.method !== 'GET') throw new Error(`arjunEndpointId ${id} não é GET (método=${ep.method})`);
        selected.push(ep);
      }
      stagesRun.push('arjun');
      for (const ep of selected) {
        if (signal.aborted) break;
        const url = buildFullUrl(api.baseUrl, ep.path);
        const result = await runArjun(url, { wordlistPath: ARJUN_WORDLIST }, { jobId, signal });
        if (result.skipped) {
          log.warn({ endpointId: ep.id, reason: result.skipped.reason }, 'arjun skipped for endpoint');
          continue;
        }
        if (result.params.length > 0) {
          await storage.appendQueryParams(ep.id, result.params.map((name) => ({ name })));
        }
      }
    } catch (err) {
      log.error({ err: String(err), apiId, stage: 'arjun' }, 'stage failed');
      stagesSkipped.push({ stage: 'arjun', reason: String(err) });
    }
  } else {
    stagesSkipped.push({ stage: 'arjun', reason: opts.dryRun ? 'dryRun' : 'disabled (opt-in)' });
  }

  if (signal.aborted) cancelled = true;
  return finalize();

  // --- Closure helpers ---
  async function finalize(): Promise<DiscoveryResult> {
    // Compute stale endpoints (in DB, not seen this run) — only when pipeline fully ran.
    let endpointsStale: string[] = [];
    try {
      const currentAll = await storage.listEndpointsByApi(apiId);
      // Stale = in-DB endpoints whose id is NOT in the set produced by any stage this run.
      // Approximation: stale = those not touched by upsert (createdAt < run start). Simpler: log all.
      // For Phase 11 (lastSeenAt deferred), we use updated_at heuristic.
      const runStart = new Date(startedAt);
      endpointsStale = currentAll.filter((e) => e.updatedAt < runStart).map((e) => e.id);
      if (endpointsStale.length > 0) await storage.markEndpointsStale(apiId, endpointsStale);
    } catch (err) {
      log.warn({ err: String(err), apiId }, 'failed to compute stale endpoints');
    }

    const result: DiscoveryResult = {
      apiId,
      stagesRun,
      stagesSkipped,
      endpointsDiscovered,
      endpointsUpdated,
      endpointsStale,
      specFetched,
      cancelled,
      durationMs: Date.now() - startedAt,
    };
    log.info({ ...result, specFetched: result.specFetched ? { url: result.specFetched.url, driftDetected: result.specFetched.driftDetected } : undefined }, 'discoverApi finished');
    return result;
  }
}

// --- Stage helpers ---

async function runSpecStage(
  api: Api,
  opts: DiscoverApiOpts,
  signal: AbortSignal,
): Promise<{ endpoints: InsertApiEndpoint[]; specFetched: DiscoveryResult['specFetched'] } | null> {
  if (api.apiType === 'graphql') {
    // GraphQL introspection — unauth first, retry if 401/403 handled inside probeGraphQL caller.
    const authHeader = await resolveCompatibleAuthHeader(api.id, api.baseUrl, opts.credentialIdOverride);
    const probe = await probeGraphQL(api.baseUrl, authHeader, signal);
    if (!probe) return null;
    const endpoints = schemaToEndpoints(probe.schema, api.id, probe.endpointPath);
    // GraphQL: specHash = hash of schema object. specVersion = 'GraphQL'.
    const { computeCanonicalHash } = await import('../scanners/api/specHash');
    const specHash = computeCanonicalHash(probe.schema);
    const driftDetected = api.specHash !== null && api.specHash !== specHash;
    if (driftDetected) log.warn({ apiId: api.id, oldHash: api.specHash, newHash: specHash }, 'spec drift detected');
    const specUrl = new URL(probe.endpointPath, api.baseUrl).toString();
    await storage.updateApiSpecMetadata(api.id, { specUrl, specVersion: 'GraphQL', specHash });
    return { endpoints, specFetched: { url: specUrl, version: 'GraphQL', hash: specHash, driftDetected } };
  }

  // REST/OpenAPI path
  let result = await fetchAndParseSpec(api.baseUrl, undefined, signal);
  if (!result) {
    // Try auth retry on 401/403 (fetchAndParseSpec returns null on any non-200; we cannot
    // distinguish 401 vs 404 here — delegating to orchestrator to retry once with cred).
    const authHeader = await resolveCompatibleAuthHeader(api.id, api.baseUrl, opts.credentialIdOverride);
    if (authHeader) {
      result = await fetchAndParseSpec(api.baseUrl, authHeader, signal);
    }
  }
  if (!result) return null;

  const driftDetected = api.specHash !== null && api.specHash !== result.specHash;
  if (driftDetected) log.warn({ apiId: api.id, oldHash: api.specHash, newHash: result.specHash }, 'spec drift detected');
  await storage.updateApiSpecMetadata(api.id, { specUrl: result.specUrl, specVersion: result.specVersion, specHash: result.specHash });
  return {
    endpoints: specToEndpoints(result.spec, api.id),
    specFetched: { url: result.specUrl, version: result.specVersion, hash: result.specHash, driftDetected },
  };
}

async function resolveCompatibleAuthHeader(
  apiId: string,
  specUrl: string,
  credIdOverride: string | undefined,
): Promise<string | undefined> {
  // Compatible for spec/graphql/httpx: api_key_header, bearer_jwt, basic, oauth2.
  const credSafe = credIdOverride
    ? await storage.getApiCredential(credIdOverride)
    : await storage.resolveApiCredential(apiId, specUrl);
  if (!credSafe) return undefined;
  const incompat = ['hmac', 'mtls', 'api_key_query'];
  if (incompat.includes(credSafe.authType)) {
    log.warn({ apiId, authType: credSafe.authType }, 'spec auth skip — incompatible cred type');
    return undefined;
  }
  const withSecret = await storage.getApiCredentialWithSecret(credSafe.id);
  if (!withSecret) return undefined;
  switch (withSecret.authType) {
    case 'api_key_header':
      return `${withSecret.headerName}: ${withSecret.secretDecrypted}`;
    case 'bearer_jwt':
      return `Bearer ${withSecret.secretDecrypted}`;
    case 'basic':
      return `Basic ${withSecret.secretDecrypted}`;
    case 'oauth2_client_credentials': {
      const token = await mintOAuth2TokenCached(withSecret);
      return token ? `Bearer ${token}` : undefined;
    }
    default:
      return undefined;
  }
}

async function buildKatanaCredential(
  apiId: string,
  baseUrl: string,
  credIdOverride: string | undefined,
): Promise<KatanaCredential | undefined> {
  const credSafe = credIdOverride
    ? await storage.getApiCredential(credIdOverride)
    : await storage.resolveApiCredential(apiId, baseUrl);
  if (!credSafe) return undefined;
  const withSecret = await storage.getApiCredentialWithSecret(credSafe.id);
  if (!withSecret) return undefined;
  switch (withSecret.authType) {
    case 'api_key_header':
      return { authType: 'api_key_header', headerName: withSecret.headerName, secret: withSecret.secretDecrypted };
    case 'bearer_jwt':
      return { authType: 'bearer_jwt', secret: withSecret.secretDecrypted };
    case 'basic':
      return { authType: 'basic', secret: withSecret.secretDecrypted };
    case 'oauth2_client_credentials':
      return { authType: 'oauth2_client_credentials', tokenUrl: withSecret.tokenUrl, clientId: withSecret.clientId, clientSecret: withSecret.clientSecret, scope: withSecret.scope };
    case 'mtls': {
      const mtls = JSON.parse(withSecret.secretDecrypted) as { cert: string; key: string; ca?: string };
      return { authType: 'mtls', credId: withSecret.id, cert: mtls.cert, key: mtls.key, ca: mtls.ca };
    }
    case 'api_key_query':
      return { authType: 'api_key_query', paramName: withSecret.paramName, secret: withSecret.secretDecrypted };
    case 'hmac':
      return { authType: 'hmac' };
    default:
      return undefined;
  }
}

// In-memory OAuth2 token cache — per-run ephemeral (drops when this function closure GC's).
const oauth2Cache = new Map<string, { token: string; expiresAt: number }>();
async function mintOAuth2TokenCached(cred: { id: string; tokenUrl: string; clientId: string; clientSecret: string; scope?: string }): Promise<string | null> {
  const now = Date.now();
  const cached = oauth2Cache.get(cred.id);
  if (cached && cached.expiresAt > now) return cached.token;
  try {
    const body = new URLSearchParams({
      grant_type: 'client_credentials',
      client_id: cred.clientId,
      client_secret: cred.clientSecret,
      ...(cred.scope ? { scope: cred.scope } : {}),
    });
    const res = await fetch(cred.tokenUrl, { method: 'POST', headers: { 'Content-Type': 'application/x-www-form-urlencoded' }, body });
    if (!res.ok) return null;
    const data = await res.json() as { access_token?: string; expires_in?: number };
    if (!data.access_token) return null;
    const ttlMs = ((data.expires_in ?? 3600) - 30) * 1000;  // minus 30s buffer per Phase 10
    oauth2Cache.set(cred.id, { token: data.access_token, expiresAt: now + ttlMs });
    return data.access_token;
  } catch {
    return null;
  }
}
```
