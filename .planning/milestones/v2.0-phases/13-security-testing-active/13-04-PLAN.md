```typescript
import 'dotenv/config';
import { parseArgs } from 'node:util';
import { runApiActiveTests } from '../services/journeys/apiActiveTests';
import type { ApiActiveTestOpts } from '@shared/schema';

async function main(): Promise<void> {
  const { values } = parseArgs({
    args: process.argv.slice(2),
    options: {
      api: { type: 'string' },
      'dry-run': { type: 'boolean', default: false },
      'no-bola': { type: 'boolean', default: false },
      'no-bfla': { type: 'boolean', default: false },
      'no-bopla': { type: 'boolean', default: false },
      'no-ssrf': { type: 'boolean', default: false },
      'rate-limit': { type: 'boolean', default: false },   // opt-in
      destructive: { type: 'boolean', default: false },
      credential: { type: 'string', multiple: true },
      help: { type: 'boolean', default: false },
    },
    strict: true,
  });

  if (values.help) {
    console.log(`
Uso: npx tsx server/scripts/runApiActiveTests.ts --api=<id> [opções]

Opções:
  --api=<id>           ID da API a testar (obrigatório)
  --dry-run            Usa fixtures locais, não faz requests reais
  --no-bola            Desabilita BOLA (API1 cross-identity)
  --no-bfla            Desabilita BFLA (API5 privilege escalation)
  --no-bopla           Desabilita BOPLA (API3 mass assignment)
  --no-ssrf            Desabilita SSRF (API7 nuclei+interactsh)
  --rate-limit         Habilita Rate-Limit absence (API4) — opt-in
  --destructive        Habilita BOPLA (requer --no-bopla=false + este flag)
  --credential=<id>    Sobrescreve credenciais (pode repetir)
  --help               Mostra esta ajuda
`.trim());
    process.exit(0);
  }

  if (!values.api) {
    console.error('Erro: --api=<id> é obrigatório');
    process.exit(1);
  }

  const opts: ApiActiveTestOpts = {
    stages: {
      bola: !(values['no-bola'] ?? false),
      bfla: !(values['no-bfla'] ?? false),
      bopla: !(values['no-bopla'] ?? false),
      ssrf: !(values['no-ssrf'] ?? false),
      rateLimit: values['rate-limit'] ?? false,
    },
    destructiveEnabled: values.destructive ?? false,
    dryRun: values['dry-run'] ?? false,
    ...(values.credential?.length ? { credentialIds: values.credential as string[] } : {}),
  };

  console.error(`[runApiActiveTests] Iniciando testes ativos para API: ${values.api}`);
  if (opts.dryRun) console.error('[runApiActiveTests] Modo dryRun — sem requests reais');
  if (!opts.destructiveEnabled) console.error('[runApiActiveTests] BOPLA desabilitado — use --destructive para habilitar');
  if (!opts.stages?.rateLimit) console.error('[runApiActiveTests] Rate-limit test desabilitado — use --rate-limit para habilitar');

  const result = await runApiActiveTests(values.api!, opts);
  console.log(JSON.stringify(result, null, 2));
}

if (import.meta.url === `file://${process.argv[1]}`) {
  main().catch((err) => {
    console.error(err);
    process.exit(1);
  });
}
```

# Testes de Segurança Ativos de API (Phase 13)

> Guia operacional para execução dos 5 vetores OWASP API Top 10 (2023) stateful:
> BOLA (API1), BFLA (API5), BOPLA (API3), Rate-Limit absence (API4), SSRF (API7).

## 1. Pré-requisitos

- Node.js ≥ 20, `npx tsx` disponível.
- `.env` configurado com `DATABASE_URL` + credenciais de acesso ao banco.
- `nuclei` instalado e templates atualizados (necessário apenas para SSRF stage):
  `nuclei -update-templates`
- Para SSRF com interactsh: `INTERACTSH_URL` env var opcional (default: `oast.me`).
  Para ambientes air-gapped: ver https://github.com/projectdiscovery/interactsh
- API alvo criada no sistema com ≥1 endpoint descoberto (Phase 11).
- ≥2 credenciais configuradas para BOLA/BFLA (Phase 10).

## 2. Segurança e Gates

**LEIA ANTES DE EXECUTAR:**

- **BOPLA (--destructive)**: Faz PUT/PATCH reais em endpoints do alvo. Use apenas em ambientes
  de desenvolvimento/staging com autorização explícita do proprietário do sistema.
  Produção: NUNCA sem autorização formal.
- **Rate-Limit (--rate-limit)**: Envia burst de 20 requests em paralelo (padrão). Pode acionar
  alertas de WAF ou rate-limiting legítimo. Use apenas em endpoints de dev/staging.
- **SSRF**: Requer acesso de saída de DNS/HTTP do appliance para `oast.me` (ou URL própria).
  Não expõe dados da aplicação — apenas injeta URL de callback.
- **dryRun (padrão)**: Seguro em qualquer ambiente — usa fixtures locais, sem requests reais.

## 3. Modo dryRun (recomendado para validar setup)

```bash

npx tsx server/scripts/runApiActiveTests.ts --api=<id> --dry-run

# Saída esperada: ActiveTestResult com findings prefixadas [DRY-RUN]
# e stagesRun: ["bola", "bfla", "ssrf"] (bopla e rate_limit skip por padrão)
```

## 5. Interpretação de Findings

| OWASP | Severity | Título padrão | Ação recomendada |
|-------|----------|---------------|-----------------|
| API1 (BOLA) | high | Acesso não autorizado a objeto via credencial secundária | Implementar ACL por objeto — verificar que principal autenticado tem permissão no objeto específico |
| API3 (BOPLA) | critical/high | Campo sensível aceito em PUT/PATCH sem validação (is_admin) | Allow-list de campos em PUT/PATCH — ignorar silenciosamente propriedades sensíveis |
| API4 (Rate-Limit) | medium | Ausência de rate-limiting em endpoint autenticado | Implementar 429 + Retry-After + X-RateLimit-* headers com limites por tier |
| API5 (BFLA) | high/medium | Privilégio administrativo acessível via credencial de baixo privilégio | RBAC em todos endpoints admin — validar privilege no backend, não confiar no cliente |
| API7 (SSRF) | varies | SSRF confirmado via interação out-of-band em parâmetro webhook_url | Allow-list de URLs destino — bloquear RFC 1918, localhost, cloud metadata (169.254.169.254) |

Findings com `[DRY-RUN]` no título são produzidas por dryRun=true — não representam vulnerabilidades reais.

## 6. Troubleshooting

**"BOLA requer ≥2 credenciais"**: Configure ≥2 credenciais distintas para a API em Phase 10.

**"nuclei preflight failed"**: Execute `nuclei -update-templates` e confirme que `nuclei` está no PATH.

**"SSRF sem interações"**: Confirme acesso de saída de DNS/HTTP para `oast.me`. Em ambientes
  air-gapped, configure `INTERACTSH_URL` apontando para servidor interactsh self-hosted.

**BOPLA não gera findings mesmo com --destructive**: Endpoints PUT/PATCH não encontrados ou
  seed body GET falhou (401/404 no recurso). Confirme que endpoints têm `requiresAuth=true`
  e credencial tem acesso ao GET equivalente.

## 7. Verificações Manuais-Only

Os seguintes comportamentos não podem ser verificados automaticamente:

| Comportamento | Motivo | Instrução |
|---------------|--------|-----------|
| BOLA finding em alvo real | Requer 2 creds com namespace de objetos compartilhado | 1. Criar 2 creds com diferentes `priority`. 2. `--api=<id> --credential=<A> --credential=<B>`. 3. Verificar finding API1 com par de requests como evidence |
| SSRF callback interactsh | Requer saída de DNS/HTTP para `oast.me` ou self-hosted | 1. Confirmar saída de rede. 2. Rodar sem `--dry-run`. 3. Aguardar até 30s. 4. Verificar finding API7 se param aceita URL |
| Rate-Limit em endpoint real | Requer endpoint que genuinamente não tem rate-limit (dev/staging) | 1. Endpoint GET+200 em ambiente sem WAF. 2. `--rate-limit`. 3. Verificar finding API4 apenas se TODOS 3 sinais ausentes |
| BOPLA gate impede writes sem --destructive | Confirmar que BOPLA não roda sem flag | 1. Rodar sem `--destructive`. 2. Verificar `stagesSkipped` contém `{ stage: 'bopla', reason: 'destructive gate not enabled' }` |
