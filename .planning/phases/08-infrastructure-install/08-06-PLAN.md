```bash
#!/usr/bin/env bash
# SamurEye Phase 8 — release tarball builder.
# Usage: build-release.sh <tag>
# Output: ./samureye-<tag>.tar.gz

set -Eeuo pipefail

TAG="${1:?Usage: $0 <tag>}"

REPO_ROOT="$(cd "$(dirname "${BASH_SOURCE[0]}")/../.." && pwd)"
WORKPARENT="$(mktemp -d)"
WORKDIR="$WORKPARENT/samureye-$TAG"
BINARIES_MANIFEST="$REPO_ROOT/scripts/install/binaries.json"
WORDLISTS_MANIFEST="$REPO_ROOT/scripts/install/wordlists.json"

RED='\033[0;31m'; GREEN='\033[0;32m'; YELLOW='\033[1;33m'; NC='\033[0m'
log()   { echo -e "${GREEN}[INFO]${NC} $*"; }
warn()  { echo -e "${YELLOW}[WARN]${NC} $*" >&2; }
error() { echo -e "${RED}[ERROR]${NC} $*" >&2; }

cleanup() { rm -rf "$WORKPARENT"; }
trap cleanup EXIT

mkdir -p "$WORKDIR/app" "$WORKDIR/bin" "$WORKDIR/wordlists"

# ── 1. App source ────────────────────────────────────────────────────────
# Use git archive if tag exists; otherwise archive HEAD (CI/dev builds).
cd "$REPO_ROOT"
if git rev-parse --verify --quiet "refs/tags/$TAG" >/dev/null; then
  log "Archiving tag $TAG"
  git archive --format=tar "$TAG" | tar -xC "$WORKDIR/app"
else
  warn "Tag $TAG não existe — usando HEAD (dev/test build)"
  git archive --format=tar HEAD | tar -xC "$WORKDIR/app"
fi

# ── 2. Download + verify each binary, save RAW ARCHIVE into bin/ ─────────
# (In --from-tarball mode, install_binary re-reads these as local file:// URLs.)
for name in $(jq -r '.binaries | keys[]' "$BINARIES_MANIFEST"); do
  url=$(jq -r ".binaries[\"$name\"].url"    "$BINARIES_MANIFEST")
  sha=$(jq -r ".binaries[\"$name\"].sha256" "$BINARIES_MANIFEST")
  fname="$(basename "$url")"
  dest="$WORKDIR/bin/$fname"

  log "Baixando $name ($url)"
  curl -fsSL --retry 2 --retry-delay 3 -o "$dest" "$url"
  actual="$(sha256sum "$dest" | awk '{print $1}')"
  if [[ "$actual" != "$sha" ]]; then
    error "Checksum mismatch para $name:"
    error "  expected=$sha"
    error "  actual=$actual"
    exit 1
  fi
  log "$name OK (sha-256 verificado)"
done

# ── 3. Wordlists (vendored + local) ───────────────────────────────────────
log "Copiando wordlists"
cp -a "$REPO_ROOT/vendor/wordlists/routes-large.kite"                    "$WORKDIR/wordlists/routes-large.kite"
cp -a "$REPO_ROOT/scripts/install/wordlists/arjun-extended-pt-en.txt"    "$WORKDIR/wordlists/arjun-extended-pt-en.txt"

# ── 4. install.sh at tarball root (user-facing entry point) ──────────────
cp -a "$WORKDIR/app/install.sh" "$WORKDIR/install.sh"

# ── 5. MANIFEST.json (binaries + wordlists with local URLs) ──────────────
# Rewrite each .binaries.<n>.url to ./bin/<basename(original_url)> and keep SHA.
log "Gerando MANIFEST.json"
jq --arg tag "$TAG" --arg date "$(date -u +%Y-%m-%dT%H:%M:%SZ)" '
  .release_tag = $tag |
  .release_date = $date |
  .binaries |= with_entries(
    .value.url = ("./bin/" + (.value.url | split("/") | last))
  )
' "$BINARIES_MANIFEST" > "$WORKDIR/MANIFEST.binaries.json"

jq '
  .wordlists |= with_entries(
    if .value.source == "local" then
      .value.url = ("./wordlists/" + .key) | .value.source = "tarball"
    elif .value.source == "remote" then
      .value.url = ("./wordlists/" + .key) | .value.source = "tarball" | del(.value.vendor_path)
    else
      .
    end
  )
' "$WORDLISTS_MANIFEST" > "$WORKDIR/MANIFEST.wordlists.json"

# Merge both into a single MANIFEST.json
jq -s '.[0] * {wordlists: .[1].wordlists}' "$WORKDIR/MANIFEST.binaries.json" "$WORKDIR/MANIFEST.wordlists.json" > "$WORKDIR/MANIFEST.json"
rm -f "$WORKDIR/MANIFEST.binaries.json" "$WORKDIR/MANIFEST.wordlists.json"

# ── 6. Tarball ────────────────────────────────────────────────────────────
OUT="$REPO_ROOT/samureye-$TAG.tar.gz"
( cd "$WORKPARENT" && tar -czf "$OUT" "samureye-$TAG" )
log "Release tarball: $OUT"
log "Tamanho: $(du -h "$OUT" | cut -f1)"
```

Find the stub function (written in Plan 04) and replace its body:

```bash
run_from_tarball() {
  local tarball="$1"
  log "═══ Modo: --from-tarball ($tarball) ═══"

  [[ -f "$tarball" ]] || { error "Tarball não encontrado: $tarball"; exit 1; }

  check_root
  detect_distro

  # 1. Extract tarball to a staging area under /opt/samureye-stage
  local stage
  stage="$(mktemp -d -t samureye-tarball-XXXXXX)"
  log "Extraindo tarball em $stage"
  tar -xzf "$tarball" -C "$stage"

  # Tarball root dir is samureye-<tag>/
  local rootdir
  rootdir="$(find "$stage" -maxdepth 1 -type d -name 'samureye-*' | head -1)"
  [[ -d "$rootdir" ]] || { error "Layout do tarball inválido"; exit 1; }

  # 2. Override MANIFEST to point at the in-tarball one — fetch_archive will read file:// URLs
  export MANIFEST="$rootdir/MANIFEST.json"

  # 3. If INSTALL_DIR doesn't exist, copy app/ into it; else safe-update flow against already-present git repo.
  if [[ ! -d "$INSTALL_DIR/.git" ]]; then
    log "Instalação fresca a partir do tarball"
    mkdir -p "$INSTALL_DIR"
    cp -a "$rootdir/app/." "$INSTALL_DIR/"
  else
    log "Tarball sobre appliance existente — usando fluxo safe-update"
    safe_reset_gate
    preserve_paths_to_staging
    rm -rf "$INSTALL_DIR"/*
    cp -a "$rootdir/app/." "$INSTALL_DIR/"
    restore_paths_from_staging
  fi

  # 4. Install binaries — install_binary will fetch from file:// URLs in MANIFEST.json
  log "Instalando binários auxiliares (offline, do tarball)..."
  # Rewrite URLs in-place: ./bin/x.zip → file://<rootdir>/bin/x.zip so fetch_archive resolves them.
  local manifest_local="$stage/MANIFEST.local.json"
  jq --arg base "$rootdir" '
    .binaries |= with_entries(
      .value.url = ("file://" + $base + "/" + (.value.url | ltrimstr("./")))
    ) |
    (.wordlists // {}) as $wl
    | .wordlists = ($wl | with_entries(
        .value.url = ("file://" + $base + "/" + (.value.url | ltrimstr("./")))
      ))
  ' "$MANIFEST" > "$manifest_local"
  export MANIFEST="$manifest_local"
  for name in $(jq -r '.binaries | keys[]' "$MANIFEST"); do
    install_binary "$name"
  done

  # 5. Install wordlists from tarball/wordlists/ — install-wordlists.sh reads the manifest.
  export WORDLISTS_MANIFEST="$MANIFEST"  # merged manifest contains both binaries and wordlists
  install_wordlists

  # 6. Rebuild app (npm install, build, migrations)
  rebuild_app

  # 7. Start service
  systemctl start "${SERVICE_NAME}" || warn "Falha ao iniciar ${SERVICE_NAME}"

  rm -rf "$stage"
  log "═══ --from-tarball concluído ═══"
}
```

Important: since the merged MANIFEST.json from build-release.sh contains both `.binaries` and `.wordlists`, re-using it as `WORDLISTS_MANIFEST` is correct — `install-wordlists.sh` only reads `.wordlists` keys.
