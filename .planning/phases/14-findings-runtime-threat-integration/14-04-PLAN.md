```typescript
// ANTES (Phase 12/13 baseline):

// }

// DEPOIS (Phase 14 Wave 3):
const result = await runApiPassiveTests(apiId, opts);  // same as before
const jobId = req.body.jobId as string | undefined;
const newFindingIds: string[] = [];
const emittedFindings: Array<JobEvent & { type: 'findings_batch' }> = [];

// Step 1: Sanitize + upsert (Phase 14 FIND-02)
let sanitizedCount = 0;
for (const finding of result.findings ?? []) {
  const originalEvidence = finding.evidence;
  try {
    finding.evidence = sanitizeApiFinding(originalEvidence);
    sanitizedCount++;
  } catch {
    // sanitizeApiFinding is fail-open internally — but double-guard here
    finding.evidence = originalEvidence;
  }
  const upserted = await upsertApiFindingByKey(finding);
  newFindingIds.push(upserted.id);
}

// Step 2: Async promotion (Phase 14 FIND-03) — fire-and-forget
let promotionKicked = false;
if (!opts.dryRun && newFindingIds.length > 0) {
  void promoteHighCriticalFindings(apiId, newFindingIds).catch((err) => {
    // eslint-disable-next-line no-console
    console.warn('[route /test/passive] promotion failed silently', {
      apiId,
      error: err instanceof Error ? err.message : String(err),
    });
  });
  promotionKicked = true;
}

// Step 3: WebSocket emit (Phase 14 FIND-04) — batch 20 per event
let eventsEmitted = 0;
if (!opts.dryRun && jobId && newFindingIds.length > 0) {
  // Fetch sanitized findings for emit payload (short fields)
  const BATCH_SIZE = 20;
  const all = result.findings ?? [];
  let batchNumber = 1;
  for (let i = 0; i < all.length; i += BATCH_SIZE) {
    const chunk = all.slice(i, i + BATCH_SIZE);
    try {
      jobEventBroadcaster.emit(jobId, {
        type: 'findings_batch',
        findings: chunk.map((f, idx) => ({
          id: newFindingIds[i + idx] ?? '',
          owaspCategory: f.owaspCategory ?? '',
          severity: f.severity ?? 'info',
          endpointPath: f.endpointPath ?? '',
          title: f.title ?? '',
        })),
        batchNumber,
        totalNewInBatch: chunk.length,
      });
      eventsEmitted++;
      batchNumber++;
    } catch {
      // jobEventBroadcaster.emit is fail-open internally; double-guard
    }
  }
}

// Response — Phase 14 extends Phase 12/13 shape additively
res.json({
  result,
  sanitized: sanitizedCount === (result.findings?.length ?? 0),
  promotionKicked,
  eventsEmitted,
});
```
